\begin{algorithmic}[1]
\Require{$\rho \geq 1$}
\Ensure{$X_k$}
\State{$NumChanges \leftarrow 1$}
\State{ $R(v) \leftarrow P(v)$}
\While{$ NumChanges >0$}
	\For{all vertex $v\in V$, \emph{\textbf Do in Parallel} }
		\State{$ r= \min (R(v), R(P(v))) $}
		\If{$r \neq R(v) $}
			\State{$R(v)= v$}
			\State{$NumChanges =NumChanges+1$}
		\EndIf
		\If{$v = R(v) $}
			\State{\emph{\textbf Report Loop} }
		\EndIf
	\EndFor
\EndWhile
\end{algorithmic}
